30 template <typename T>
31 void LockFreeQueue<T>::Push(const T& value)
32 {
33 Node* new_node = new Node(value); // Allocate new node with given value
34
35 while (true)
36 {
37 // Load the current tail node (last node in the queue)
38 Node* last = m_tail.load(std::memory_order_acquire);
39
40 // Load the next pointer of last node to see if someone else inserted after it
41 Node* next = last->mNext.load(std::memory_order_acquire);
42
43 // Check if tail pointer is still consistent after loading next
44 if (last == m_tail.load(std::memory_order_acquire))
45 {
46 if (next == nullptr) // Tail is pointing at actual last node (no next)
47 {
48 // Try to link the new node as the next node of last (push)
49 if (last->mNext.compare_exchange_weak(next, new_node,
50 std::memory_order_release,
51 std::memory_order_relaxed))
52 {
53 // Try to swing the tail pointer forward to the new node
54 m_tail.compare_exchange_weak(last, new_node,
55 std::memory_order_release,
56 std::memory_order_relaxed);
57
58 return; // Push successful
59 }
60 }
61 else
62 {
63 // Another thread inserted a node after last but tail pointer lagged behind
64 // Move tail forward to catch up
65 m_tail.compare_exchange_weak(last, next,
66 std::memory_order_release,
67 std::memory_order_relaxed);
68 }
69 }
70 }
71 }

73 template <typename T>
74 T* LockFreeQueue<T>::Pop()
75 {
76 thread_local ThreadCleanup cleanupGuard; // Ensures retired nodes are cleaned up periodically when thread exits
77
78 HazardPointerGuard guard; // Acquire hazard pointer for current thread
79 HazardPointer* hp = guard.Get(); // Get the hazard pointer slot to protect nodes
80
81 while (true)
82 {
83 Node* first;
84
85 // Load the head pointer, protect it with hazard pointer, then verify no change
86 do {
87 first = m_head.load(std::memory_order_acquire);
88 hp->mPtr.store(first, std::memory_order_release);
89 } while (first != m_head.load(std::memory_order_acquire));
90
91 // Load next pointer (the node after the dummy or current head)
92 Node* next = first->mNext.load(std::memory_order_acquire);
93
94 if (!next)
95 {
96 // Queue is empty (only dummy node), return nullptr to signal empty
97 return nullptr;
98 }
99
100 // Copy the data from next node to return
101 T* result = new T(next->mData);
102
103 // Attempt to swing head forward from first to next atomically
104 if (m_head.compare_exchange_weak(first, next,
105 std::memory_order_release,
106 std::memory_order_relaxed))
107 {
108 // Success: Clear hazard pointer protection on old head node
109 hp->mPtr.store(nullptr, std::memory_order_release);
110
111 // Retire the old head node (dummy node) for deferred reclamation
112 HazardPointerManager::GetInstance().RetireNode(first, ReclaimNode);
113
114 return result; // Return copied data pointer
115 }
116
117 // CAS failed: some other thread modified head first; rollback
118 delete result;
119 }
120 }
